# Khai báo một biến để chứa giá trị các thuộc tính, sau có thể tái sử dụng trong thẻ trong JSX:
import React from 'react';

const redPanda = {
  src: 'https://upload.wikimedia.org/wikipedia/commons/b/b2/Endangered_Red_Panda.jpg',
  alt: 'Red Panda',
  width:  '200px'
};

function RedPanda(){
    return (
      <div>
        <h1>Cute Red Panda</h1>
        <img 
          src={redPanda.src}
          alt={redPanda.alt}
          width={redPanda.width} />
      </div>
    );
}

export default RedPanda;
- Trong ví dụ này, biến RedPanda chứa thuộc tính, được sử dụng khi viết thẻ img
